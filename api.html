POST /api/developer/seraphina/chat

Seraphina Chat (Research)

Access the Seraphina model, powered by Perplexity Sonar Pro. This model excels at deep reasoning, web grounding, and providing citations. It is ideal for complex queries requiring up-to-date information.

Body Parameters

  • message string The user's query or prompt.
  • history array List of previous messages (role/content).
  • temperature float Creativity (0.0 to 1.0). Default: 0.7.
  • stream boolean Set to true for SSE streaming. 
import requests

url = "https://web-production-9a18.up.railway.app/api/developer/seraphina/chat"

headers = {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json"
}

payload = {
    "message": "Explain quantum entanglement simply.",
    "history": [
        {"role": "user", "content": "Hello"},
        {"role": "assistant", "content": "Hi! How can I help?"}
    ],
    "temperature": 0.7,
    "stream": False 
}

response = requests.post(url, json=payload, headers=headers)
print(response.json())

POST /api/developer/bobby/chat

Bobby Chat (Fast)

Interact with Bobby (powered by Thena Llama). This model is cost-effective, faster, and optimized for creative writing, chatting, and general tasks without web search requirements.

Body Parameters

  • prompt string The input message.
  • systemPrompt string Optional. Override the default persona. 
import requests

url = "https://web-production-9a18.up.railway.app/api/developer/bobby/chat"

payload = {
    "prompt": "Write a short poem about coding.",
    "systemPrompt": "You are a poetic assistant."
}
headers = { "Authorization": "Bearer YOUR_API_KEY" }

response = requests.post(url, json=payload, headers=headers)
print(response.json())
const axios = require('axios');

axios.post('https://web-production-9a18.up.railway.app/api/developer/bobby/chat', {
    prompt: "Hello Bobby!",
}, {
    headers: { "Authorization": "Bearer YOUR_API_KEY" }
}).then(res => console.log(res.data));
curl -X POST https://web-production-9a18.up.railway.app/api/developer/bobby/chat \
-H "Authorization: Bearer YOUR_API_KEY" \
-H "Content-Type: application/json" \
-d '{"prompt": "Hello"}'

POST /api/developer/generate-image

Generate Image

Create high-fidelity AI art using Thena models. Supports various styles including photorealism, anime, and v6 illustration.

Body Parameters

  • prompt string Detailed description of the image.
  • model string Options: v6, photoreal, anime_core.
  • width/height int Optional. e.g., 1024. 
import requests
import base64

url = "https://web-production-9a18.up.railway.app/api/developer/generate-image"

payload = {
    "prompt": "Cyberpunk city street at night, neon rain",
    "model": "photoreal",
    "width": 1024,
    "height": 1024
}
headers = { "Authorization": "Bearer YOUR_API_KEY" }

response = requests.post(url, json=payload, headers=headers)
data = response.json()

# Decode Base64
if "image_b64" in data:
    with open("output.png", "wb") as f:
        f.write(base64.b64decode(data["image_b64"]))
const axios = require('axios');
const fs = require('fs');

const url = "https://web-production-9a18.up.railway.app/api/developer/generate-image";
const headers = { "Authorization": "Bearer YOUR_API_KEY" };

axios.post(url, {
    prompt: "Anime style warrior, epic lighting",
    model: "anime_core"
}, { headers }).then(res => {
    const buffer = Buffer.from(res.data.image_b64, 'base64');
    fs.writeFileSync('result.png', buffer);
});
